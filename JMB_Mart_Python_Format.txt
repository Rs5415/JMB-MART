# --- SECURITY & AUTH MIDDLEWARE ---

async def get_current_user(authorization: str = Header(None)):
    if not authorization:
        raise HTTPException(status_code=401, detail=\"Auth token required\")
    try:
        # Verify Firebase Token
        token = authorization.replace(\"Bearer \", \"\")
        decoded_token = auth.verify_id_token(token)
        return decoded_token
    except Exception as e:
        raise HTTPException(status_code=401, detail=f\"Invalid token: {str(e)}\")

async def admin_only(user=Depends(get_current_user)):
    user_doc = db.collection(\"users\").document(user['uid']).get()
    if not user_doc.exists:
        # Check if they are the hardcoded super-admin
        if user.get(\"email\") == ADMIN_EMAIL:
            return user
        raise HTTPException(status_code=403, detail=\"User record not found\")
    
    data = user_doc.to_dict()
    if data.get(\"email\") == ADMIN_EMAIL or data.get(\"role\") == \"admin\":
        return user
    raise HTTPException(status_code=403, detail=\"Admin access required\")

@app.get(\"/api/admin/stats\", response_model=Dict[str, Any])
async def get_admin_dashboard_stats(_=Depends(admin_only)):
    \"\"\"Calculates all metrics for the Admin Dashboard Home\"\"\"
    orders = db.collection(\"orders\").stream()
    users = db.collection(\"users\").stream()
    
    order_list = [o.to_dict() for o in orders]
    user_list = [u.to_dict() for u in users]
    
    revenue = sum(o.get(\"total\", 0) for o in order_list if o.get(\"status\") == \"delivered\")
    pending = len([o for o in order_list if o.get(\"status\") == \"pending\"])
    delivered = len([o for o in order_list if o.get(\"status\") == \"delivered\"])
    
    return {
        \"total_orders\": len(order_list),
        \"revenue\": revenue,
        \"pending_orders\": pending,
        \"delivered_orders\": delivered,
        \"total_users\": len(user_list),
        \"success_rate\": (delivered / len(order_list) * 100) if order_list else 0
    }

@app.get(\"/api/admin/analytics/sales\")
async def get_sales_analytics(_=Depends(admin_only)):
    \"\"\"Generates data for the 7-day sales line chart\"\"\"
    now = datetime.now()
    sales_data = []
    
    for i in range(6, -1, -1):
        day = now - timedelta(days=i)
        start_of_day = datetime(day.year, day.month, day.day)
        end_of_day = start_of_day + timedelta(days=1)
        
        daily_orders = db.collection(\"orders\") \\
            .where(\"status\", \"==\", \"delivered\") \\
            .where(\"createdAt\", \">=\", start_of_day) \\
            .where(\"createdAt\", \"<\", end_of_day) \\
            .stream()
            
        total = sum(o.to_dict().get(\"total\", 0) for o in daily_orders)
        sales_data.append({
            \"name\": day.strftime(\"%d/%m\"),
            \"total\": total
        })
    return sales_data

@app.post(\"/api/admin/products\")
async def create_product(product: Product, _=Depends(admin_only)):
    doc_ref = db.collection(\"products\").document()
    product_data = product.dict(exclude={\"id\"})
    product_data[\"createdAt\"] = firestore.SERVER_TIMESTAMP
    doc_ref.set(product_data)
    return {\"id\": doc_ref.id, \"status\": \"Product Success\"}
